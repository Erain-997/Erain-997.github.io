```go
var count int

type heart struct {
}

// 知识点：定时任务
func Test0208(t *testing.T) {
	h := heart{}
	h.HeartbeatFunc()
	time.Sleep(time.Hour) // 主程序运行
}

// 定时任务快捷写法
func (h heart) HeartbeatFunc() {
	a := time.AfterFunc(4*time.Second, h.HeartbeatFunc) // 定时任务  自带子线程
	count++
	fmt.Println(time.Now(), count) // 打印下心跳次数
	// 设置停止条件
	if count == 5 {
		a.Stop()
	}
}
```
